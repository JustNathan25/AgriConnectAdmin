rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // =============================
    // üîπ HELPER FUNCTIONS
    // =============================
    
    function isSignedIn() {
      return request.auth != null;
    }
    
    function isAdmin() {
      return request.auth != null 
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    function isOwner(userId) {
      return request.auth != null && request.auth.uid == userId;
    }
    
    function isParticipant(chatData) {
      return request.auth != null && 
             request.auth.uid in chatData.participants;
    }

    /* ===========================
       üë§ USERS COLLECTION
    =========================== */
    match /users/{userId} {
      // Users can read all, create their own, update own profile
      allow read: if isSignedIn();
      allow create: if isSignedIn() && isOwner(userId);
      
      // Users can update their own profile OR admins can update (for warnings)
      allow update: if isOwner(userId) || isAdmin();
      
      allow delete: if isAdmin();
      
      // User's reviews subcollection (legacy)
      match /reviews/{reviewId} {
        allow read: if isSignedIn();
        allow write: if isSignedIn();
        allow delete: if isOwner(userId) || isAdmin();
      }
      
      // ‚≠ê NEW REVIEW SYSTEM - Reviews received by this user
      match /reviews_received/{reviewId} {
        allow read: if isSignedIn();
        allow write: if isSignedIn();
        allow delete: if isOwner(userId) || isAdmin();
      }
      
      // ‚≠ê NEW REVIEW SYSTEM - Reviews given by this user
      match /reviews_given/{reviewId} {
        allow read: if isSignedIn();
        allow write: if isOwner(userId);
        allow delete: if isOwner(userId) || isAdmin();
      }
      
      // ‚≠ê NEW REVIEW SYSTEM - Rating summary/statistics
      match /rating_summary/{summaryId} {
        allow read: if isSignedIn();
        allow write: if isSignedIn();
        allow delete: if isAdmin();
      }
      
      // üîî User notifications subcollection
      match /notifications/{notificationId} {
        allow read: if isOwner(userId);
        allow write: if isSignedIn();
        allow delete: if isOwner(userId) || isAdmin();
      }
      
      // üåæ User's crops subcollection
      match /crops/{cropId} {
        allow read: if isSignedIn();
        allow write: if isOwner(userId);
        allow delete: if isOwner(userId) || isAdmin();
      }
      
      // üåæ User's farm showcases subcollection
      match /showcases/{showcaseId} {
        allow read: if isSignedIn();
        allow write: if isOwner(userId);
        allow delete: if isOwner(userId) || isAdmin();
      }
      
      // ‚ù§Ô∏è User's likes subcollection
      match /likes/{likeId} {
        allow read: if isSignedIn();
        allow write: if isOwner(userId);
        allow delete: if isOwner(userId);
      }
      
      // ‚ù§Ô∏è User's liked products
      match /likedProducts/{productId} {
        allow read: if isSignedIn();
        allow write: if isOwner(userId);
        allow delete: if isOwner(userId);
      }
    }

    /* ===========================
       üõí PRODUCTS COLLECTION
    =========================== */
    match /products/{productId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn();
      
      // Allow update by owner or admin only
      allow update: if isSignedIn() && (
        request.auth.uid == resource.data.sellerId || isAdmin()
      );
      
      allow delete: if isAdmin() 
                    || (isSignedIn() && request.auth.uid == resource.data.sellerId);

      match /reviews/{reviewId} {
        allow read: if isSignedIn();
        allow write: if isSignedIn();
        allow delete: if isSignedIn() || isAdmin();
      }
    }

    /* ===========================
       üí¨ CHATS COLLECTION (FIXED - SECURE)
    =========================== */
    match /chats/{chatId} {
      // Read: Users can read chats they participate in, or admins can read all
      allow read: if isSignedIn() 
                  && (isParticipant(resource.data) || isAdmin());
      
      // Create: Admins can create system chats, users can create regular chats where they are participants
      allow create: if isSignedIn() && (
        // Admin can create any chat (system or regular)
        isAdmin() ||
        // Non-admin can only create non-system chats where they are a participant
        (
          request.resource.data.isAdminChat != true &&
          request.auth.uid in request.resource.data.participants
        )
      );
      
      // Update: Same restrictions as create
      allow update: if isSignedIn() && (
        isAdmin() ||
        (
          resource.data.isAdminChat != true &&
          request.auth.uid in resource.data.participants
        )
      );
      
      allow delete: if isAdmin();
      
      match /messages/{messageId} {
        // Read: Users can read messages in chats they participate in, or admins
        allow read: if isSignedIn() 
                    && (isParticipant(get(/databases/$(database)/documents/chats/$(chatId)).data) || isAdmin());
        
        // Create: CRITICAL SECURITY - Only admins can create system messages
        allow create: if request.auth != null && (
          (
            // System messages require admin privilege
            request.resource.data.system == true &&
            isAdmin()
          ) ||
          (
            // Regular messages can be created by participants
            request.resource.data.system != true &&
            request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants
          )
        );
        
        // Update: Only users can update their own non-system messages (e.g., mark as read)
        allow update: if request.auth != null && (
          resource.data.system != true &&
          request.auth.uid == resource.data.senderId
        );
        
        // Delete: Only admins can delete messages
        allow delete: if isAdmin();
      }
    }

    /* ===========================
       üö® REPORTS COLLECTION (Chat Reports)
    =========================== */
    match /reports/{reportId} {
      allow create: if isSignedIn();
      allow read: if isAdmin() || isOwner(resource.data.reportedBy);
      allow update, delete: if isAdmin();
    }

    /* ===========================
       üé´ TICKETS COLLECTION (Product/Profile Reports)
    =========================== */
    match /tickets/{ticketId} {
      allow create: if isSignedIn();
      allow read: if isAdmin() || isOwner(resource.data.reportedBy);
      allow update, delete: if isAdmin();
    }

    /* ===========================
       ‚≠ê REVIEWS COLLECTION (Master)
    =========================== */
    match /reviews/{reviewId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn();
      allow update: if isSignedIn() 
                    && request.auth.uid == resource.data.reviewerUserId;
      allow delete: if isAdmin() 
                    || (isSignedIn() && request.auth.uid == resource.data.reviewerUserId);
    }
    
    /* ===========================
       ‚≠ê REVIEW REPORTS COLLECTION
    =========================== */
    match /review_reports/{reportId} {
      allow create: if isSignedIn();
      allow read: if isAdmin() || isOwner(resource.data.reportedBy);
      allow update, delete: if isAdmin();
    }

    /* ===========================
       üîî NOTIFICATIONS COLLECTION (Global)
    =========================== */
    match /notifications/{notificationId} {
      // Users can read their own notifications
      allow read: if isSignedIn() && isOwner(resource.data.userId);
      
      // Allow creation by signed-in users (for messages, reviews, etc.)
      allow create: if isSignedIn();
      
      // Users can update 'read' and 'seen' fields on their own notifications
      allow update: if isSignedIn() 
                    && isOwner(resource.data.userId)
                    && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['read', 'seen']);
      
      // Users can delete their own notifications, admins can delete any
      allow delete: if isOwner(resource.data.userId) || isAdmin();
    }

    /* ===========================
       üîî ADMIN NOTIFICATIONS COLLECTION
    =========================== */
    match /admin_notifications/{notificationId} {
      // Only admins can read admin notifications
      allow read: if isAdmin();
      
      // Users can create notifications (when submitting payment proof)
      // Admins can create notifications
      allow create: if isSignedIn();
      
      // Only admins can update notifications (mark as read, etc.)
      allow update: if isAdmin();
      
      // Only admins can delete notifications
      allow delete: if isAdmin();
    }

    /* ===========================
       ‚ö†Ô∏è USER WARNINGS COLLECTION
    =========================== */
    match /user_warnings/{warningId} {
      // Users can read their own warnings
      allow read: if isSignedIn() && isOwner(resource.data.userId);
      
      // Only admins can create warnings
      allow create: if isAdmin();
      
      // Users can acknowledge warnings (update acknowledged and acknowledgedAt only)
      // Admins can update anything
      allow update: if isAdmin() 
                    || (isSignedIn() 
                        && isOwner(resource.data.userId)
                        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['acknowledged', 'acknowledgedAt']));
      
      // Only admins can delete warnings
      allow delete: if isAdmin();
    }

    /* ===========================
       üì¶ PRODUCT WARNINGS COLLECTION
    =========================== */
    match /product_warnings/{warningId} {
      // Sellers can read warnings for their products
      allow read: if isSignedIn() && isOwner(resource.data.sellerId);
      
      // Only admins can create product warnings
      allow create: if isAdmin();
      
      // Sellers can acknowledge warnings (update acknowledged and acknowledgedAt only)
      // Admins can update anything
      allow update: if isAdmin() 
                    || (isSignedIn() 
                        && isOwner(resource.data.sellerId)
                        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['acknowledged', 'acknowledgedAt']));
      
      // Only admins can delete warnings
      allow delete: if isAdmin();
    }

    /* ===========================
       üóëÔ∏è PRODUCT DELETIONS COLLECTION (Admin Audit Log)
    =========================== */
    match /product_deletions/{deletionId} {
      // Only admins can read, write, and delete
      allow read, write, delete: if isAdmin();
    }

    /* ===========================
       üìã VERIFICATIONS COLLECTION
    =========================== */
    match /verifications/{userId} {
      // Users can create their own verification request
      allow create: if isSignedIn() && isOwner(userId);
      
      // Users can read their own verification
      // Admins can read all verifications
      allow read: if isAdmin() || isOwner(userId);
      
      // Only admins can update verifications (approve/reject)
      allow update: if isAdmin();
      
      // Only admins can delete verifications
      allow delete: if isAdmin();
    }

    /* ===========================
       üêõ APP REPORTS COLLECTION
    =========================== */
    match /app_reports/{reportId} {
      // Users can create app reports
      allow create: if isSignedIn();
      
      // Admins can read all reports
      // Users can read their own reports
      allow read: if isAdmin() || (isSignedIn() && isOwner(resource.data.userId));
      
      // Only admins can update app reports
      allow update: if isAdmin();
      
      // Only admins can delete app reports
      allow delete: if isAdmin();
    }

    /* ===========================
       üí∞ PAYMENT TRANSACTIONS COLLECTION (FIXED - camelCase)
    =========================== */
    match /paymentTransactions/{transactionId} {
      // Users can read their own transactions
      // Admins can read all transactions
      allow read: if isSignedIn() 
                  && (isOwner(resource.data.userId) || isAdmin());
      
      // Users can create their own payment transactions
      allow create: if isSignedIn() && isOwner(request.resource.data.userId);
      
      // Only admins can update transactions (verify, approve, reject)
      allow update: if isAdmin();
      
      // Only admins can delete transactions
      allow delete: if isAdmin();
    }

    /* ===========================
       üöÄ BOOST HISTORY COLLECTION
    =========================== */
    match /boost_history/{historyId} {
      // Users can read their own boost history
      // Admins can read all
      allow read: if isSignedIn() 
                  && (isOwner(resource.data.userId) || isAdmin());
      
      // Only admins can create boost history (when approving payment)
      allow create: if isAdmin();
      
      // Only admins can update boost history
      allow update: if isAdmin();
      
      // Only admins can delete boost history
      allow delete: if isAdmin();
    }

    /* ===========================
       üí≥ PAYMENTS COLLECTION (Generic - if exists)
    =========================== */
    match /payments/{paymentId} {
      allow read: if isSignedIn() 
                  && (isOwner(resource.data.userId) || isAdmin());
      allow create: if isSignedIn();
      allow update, delete: if isAdmin();
    }

    /* ===========================
       üîç COLLECTION GROUP ACCESS
    =========================== */
    match /{path=**}/reviews/{reviewId} {
      allow read: if isSignedIn();
      allow write: if isSignedIn();
    }
    
    match /{path=**}/notifications/{notificationId} {
      allow read: if isSignedIn();
      allow write: if isSignedIn();
    }
  }
}
